template <typename T>
shared_ptr&lt;node<T>> zipping_linked_list(const shared_ptr&lt;node<T>> &amp; L) {
    // Find the middle point of the linked list
    shared_ptr&lt;node<T>> slow = L, fast = L, pre_slow = nullptr;
    while (fast) {
        fast = fast->next;
        if (fast) {
            pre_slow = slow;
            slow = slow->next;
            fast = fast->next;
        }
    }
    // The list only contains one node
    if (!pre_slow) return L;
    // Split the list into two lists
    pre_slow->next = nullptr;
    shared_ptr&lt;node<T>> reverse = reversed_linked_list(slow), curr = L;
    // Zipping the list
    while (curr &amp;&amp; reverse) {
        connect_a_next_to_b_advance_a(curr, reverse);
        if (curr) connect_a_next_to_b_advance_a(reverse, curr);
    }
    return L;
}
